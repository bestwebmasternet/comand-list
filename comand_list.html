Git Comand:

git --version - перевірка версії git
git config --list - переглянути налоштування
git config --global user.name "Sasha Zimin" - зміна імені автора
git config --global user.email "info@zimin.dev" - вказуємо пошту автора
git init - створення нового репозиторія
git config --global init.defaulBranch 'name' - зміна головного імені гілки (master, main)
git status - переглянути файли, які знаходяться в індексі (файли що мали зміни, але ще не закомічені з робочої дерикторії)
git add file - додавання всіх змін в індекс (додавання змін)
git commit - збереження з індексу в репозиторій (збереження змін)
git commit -m "message" - створення та збереження нового ко
git checkout "comit hash" - перейти до певної версії коміта по хешу (хеш можна подивитися в логах)
git log - переглянути логи зміни комітів
git checkout "branch name" - перейти в певну гілку в останвй коміт
git checkout "commit hash" - перейти в певний коміт, що був раніше (переміщує HEAD до цього коміту)
git checkout main (або master) - перейти в основну гілку main та зафіксувати HEAD над останнім комітом
git checkout назвагілки - перейти в гілку, переміщує HEAD на вибрану гілку
git cat-file - t 2c4f323 (hash) - виводить інфо про файл, тип файлу: коміт, дерево
git cat-file - p 2c4f323 - виводить інфо про автора коміта або дерева по хешу, та які зміни були в файлі
git status - виводить іфо стану репозиторія зараз
git branch - виводить перелік усіх гілок
git branch названовоїгілки - створює нову гілку
git checkout -b nazvanovoigilku - створює нову гілку та переходить в неї
git branch -m novanazvagilku - перейменовує гілку
git branch -d nazvagilku - видалення певної гілки *щоб видалити певну гілку, з неї потрібно спочатку вийти
git merge nazvagilki яку ми хочено злити з іншою гілкою *команду виконуємо перебуваючі в тій гілці(receiving branch), з якою ми хочемо злити певну гілку (feature branch)
code . - команада з терміналу? проект який в роботі зараз відкриє в vscode * в vscode можна перемикаьтися поміж гілок в нижньому лівому куті
